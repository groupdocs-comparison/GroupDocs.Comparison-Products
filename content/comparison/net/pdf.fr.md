            // Ajouter une ou plusieurs PDF s à la comparaison

            comparer.Add("file_to_compare_2.pdf");
            comparer.Add("file_to_compare_3.pdf");

            // Rapport de résultats à analyser
            comparer.Compare("result.pdf"); 
        }
